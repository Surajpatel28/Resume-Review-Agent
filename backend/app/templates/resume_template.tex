% Custom commands

% Resume bullet item — no extra negative vspace
\newcommand{\resumeItem}[1]{
  \item\small{#1}
}

% Education / Experience heading — uses p{} column for wrapping
\newcommand{\resumeSubheading}[4]{
  \item
    \begin{tabularx}{0.95\textwidth}{@{}p{0.75\textwidth} r@{}}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabularx}\vspace{-3pt}
}

% Project heading — uses X column for auto-wrapping long titles
\newcommand{\resumeProjectHeading}[2]{%
    \item
    \begin{tabularx}{0.95\textwidth}{@{}X r@{}}
      \small\textbf{#1} & #2 \\
    \end{tabularx}\vspace{-2pt}
}

% Tech stack on a separate line under project heading — parbox prevents overflow
\newcommand{\resumeProjectTech}[1]{%
    \parbox[t]{0.95\textwidth}{\textit{\small #1}}\vspace{-2pt}
}
